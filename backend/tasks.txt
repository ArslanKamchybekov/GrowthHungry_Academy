@Put('/update-role/:id')
    async updateRole(@Param ('id')id: string, @Body ()user: IUser) { 
        try {
            const updatedUser = await this.userService.updateRole(id, user.role);
            return updatedUser;
        } catch (error) {
            return { error: error.message };
        }
    }

async updateRole(id: string, role: string) {
